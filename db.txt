//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table users as U {
  id int
	surname varchar
	name varchar
  email varchar
  password varchar
  admin tinyint
  change_pass tinyint
}

Table cases {
  id int
	created_by id
	title varchar
  desc varchar
  category_id int
  created_at timestamp
  deleted tinyint
  modified_at timestamp
  variant id
}

Table rappresentations{
  id_case int
  date datetime
}

Table categories{
  id int
  name varchar
}

Ref: "cases"."created_by" < "users"."id"

Ref: "cases"."id" < "rappresentations"."id_case"

Ref: "categories"."id" < "cases"."category_id"


Ref: "cases"."variant" < "cases"."id"
